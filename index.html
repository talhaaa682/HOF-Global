// ══════════════════════════════════════════════════════════
// HOF COMPLIANCE — Apps Script  (replace your full script)
// Auto-detects column headers — no manual counting needed
// ══════════════════════════════════════════════════════════

// ▶ UPDATE THIS to match your exact sheet tab name:
var RAW_SHEET_NAME = 'RAW AUDIT NEW';

// Header → internal field name mapping (add variants if your headers differ)
var HEADER_MAP = {
  'consultant name': '_name', 'agent name': '_name', 'name': '_name', 'consultant': '_name',
  'case link': '_link', 'hubspot link': '_link', 'link': '_link', 'url': '_link',
  'deal link': '_link', 'contact link': '_link', 'case url': '_link',
  'contact name': '_cname', 'case name': '_cname', 'lead name': '_cname', 'client name': '_cname',
  'date': '_date', 'audit date': '_date',
  '1st contact': 'firstContact', 'first contact': 'firstContact',
  '1st call': 'firstCall', 'first call': 'firstCall', '1st call made': 'firstCall',
  'call logged': 'callLogged',
  'outcome': 'outcome',
  'call desc': 'callDesc', 'call description': 'callDesc',
  'lead stage': 'leadStage', 'lead stage updated': 'leadStage', 'lead stage upd': 'leadStage',
  'correct stage': 'correctStage',
  'qualified mark': 'qualifiedMark', 'qualified': 'qualifiedMark',
  'deal created': 'deal', 'deal': 'deal',
  'pipeline': 'pipeline',
  'timeline': 'timeline',
  'properties': 'properties',
  'email sent': 'emailSent',
  'email via hub': 'emailHub', 'email via hubspot': 'emailHub',
  'prof tone': 'profTone', 'professional tone': 'profTone',
  'signature': 'signature',
  'wa used': 'waUsed', 'whatsapp used': 'waUsed',
  'wa logged': 'waLogged', 'whatsapp logged': 'waLogged',
  'wa note': 'waNote', 'whatsapp note': 'waNote',
  'task created': 'taskCreated',
  'task done': 'taskDone',
  'task type': 'taskType',
};

var ERROR_FIELDS = [
  'firstContact','firstCall','callLogged','outcome','callDesc',
  'leadStage','correctStage','qualifiedMark',
  'deal','pipeline','timeline','properties',
  'emailSent','emailHub','profTone','signature',
  'waUsed','waLogged','waNote',
  'taskCreated','taskDone','taskType'
];

function doGet(e) {
  try {
    var params = e ? e.parameter : {};
    var startStr = params.start || '';
    var endStr   = params.end   || '';
    var startDate = startStr ? new Date(startStr) : null;
    var endDate   = endStr   ? new Date(endStr)   : null;
    if (endDate) endDate.setHours(23, 59, 59);

    var ss        = SpreadsheetApp.getActiveSpreadsheet();
    var rawSheet  = ss.getSheetByName(RAW_SHEET_NAME);

    // Build errorLinks from raw audit sheet
    var errorLinks = rawSheet ? buildErrorLinks(rawSheet, startDate, endDate) : {};

    // ── Build your aggregated data here (keep your existing logic) ──────
    // This example reads from the same raw sheet and aggregates per consultant.
    // If you already have a separate aggregation sheet/function, merge errorLinks there.
    var agentData = aggregateFromRawSheet(rawSheet, startDate, endDate, errorLinks);

    var output = {
      ok: true,
      data: agentData,
      totalAudits: agentData.reduce(function(s,d){ return s + d.audits; }, 0),
      dateRange: { min: startStr || 'All time', max: endStr || 'present' }
    };

    return ContentService
      .createTextOutput(JSON.stringify(output))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ ok: false, error: err.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// ── Build errorLinks map from raw audit sheet ─────────────────────────────
function buildErrorLinks(sheet, startDate, endDate) {
  var data    = sheet.getDataRange().getValues();
  var headers = data[0].map(function(h){ return String(h||'').trim().toLowerCase(); });

  // Auto-detect column indices from headers
  var col = {};
  headers.forEach(function(h, i) {
    var mapped = HEADER_MAP[h];
    if (mapped && col[mapped] === undefined) col[mapped] = i;
  });

  // Fuzzy fallback for name and link columns
  if (col['_name'] === undefined) {
    headers.forEach(function(h,i){
      if ((h.includes('name') || h.includes('consultant') || h.includes('agent')) && col['_name']===undefined) col['_name'] = i;
    });
  }
  if (col['_link'] === undefined) {
    headers.forEach(function(h,i){
      if ((h.includes('link') || h.includes('url') || h.includes('hubspot') || h.includes('http')) && col['_link']===undefined) col['_link'] = i;
    });
  }

  var errorLinks = {};

  for (var i = 1; i < data.length; i++) {
    var row  = data[i];
    var name = String(row[col['_name']] || '').trim();
    if (!name) continue;

    // Date filter
    if (startDate || endDate) {
      var d = new Date(row[col['_date']]);
      if (isNaN(d)) continue;
      if (startDate && d < startDate) continue;
      if (endDate   && d > endDate)   continue;
    }

    var link = {
      url:  col['_link']  !== undefined ? String(row[col['_link']]  || '').trim() : '',
      name: col['_cname'] !== undefined ? String(row[col['_cname']] || '').trim() : ('Row ' + (i + 1)),
      date: col['_date']  !== undefined && row[col['_date']]
              ? Utilities.formatDate(new Date(row[col['_date']]), Session.getScriptTimeZone(), 'dd MMM yyyy')
              : ''
    };
    if (!link.name) link.name = 'Row ' + (i + 1);

    if (!errorLinks[name]) errorLinks[name] = {};

    ERROR_FIELDS.forEach(function(field) {
      if (col[field] === undefined) return;
      var val = String(row[col[field]] || '').trim().toLowerCase();
      var isErr = (val === '1' || val === 'yes' || val === 'x' || val === 'true');
      if (isErr) {
        if (!errorLinks[name][field]) errorLinks[name][field] = [];
        errorLinks[name][field].push(link);
      }
    });
  }

  return errorLinks;
}

// ── Aggregate per-consultant stats from raw sheet ────────────────────────
// (This replaces your existing aggregation — adapt as needed)
function aggregateFromRawSheet(sheet, startDate, endDate, errorLinks) {
  var data    = sheet.getDataRange().getValues();
  var headers = data[0].map(function(h){ return String(h||'').trim().toLowerCase(); });

  var col = {};
  headers.forEach(function(h, i) {
    var mapped = HEADER_MAP[h];
    if (mapped && col[mapped] === undefined) col[mapped] = i;
  });
  if (col['_name'] === undefined) {
    headers.forEach(function(h,i){
      if ((h.includes('name')||h.includes('consultant')||h.includes('agent')) && col['_name']===undefined) col['_name']=i;
    });
  }

  var agents = {};

  for (var i = 1; i < data.length; i++) {
    var row  = data[i];
    var name = String(row[col['_name']] || '').trim();
    if (!name) continue;

    if (startDate || endDate) {
      var d = new Date(row[col['_date']]);
      if (isNaN(d)) continue;
      if (startDate && d < startDate) continue;
      if (endDate   && d > endDate)   continue;
    }

    if (!agents[name]) {
      agents[name] = { name: name, audits: 0, total: 0 };
      ERROR_FIELDS.forEach(function(f){ agents[name][f] = 0; });
    }
    agents[name].audits++;

    ERROR_FIELDS.forEach(function(field) {
      if (col[field] === undefined) return;
      var val = String(row[col[field]] || '').trim().toLowerCase();
      if (val === '1' || val === 'yes' || val === 'x' || val === 'true') {
        agents[name][field]++;
        agents[name].total++;
      }
    });
  }

  // Calculate compliance % and attach errorLinks
  return Object.values(agents).map(function(a) {
    var maxPts = a.audits * 100;
    a.pct = maxPts > 0 ? Math.max(0, Math.round(((maxPts - a.total) / maxPts) * 100)) : 100;
    a.catErr_outcome   = a.outcome || 0;
    a.catErr_call      = (a.firstContact||0)+(a.firstCall||0)+(a.callLogged||0)+(a.callDesc||0);
    a.catErr_leadStage = (a.leadStage||0)+(a.correctStage||0)+(a.qualifiedMark||0);
    a.catErr_deal      = (a.deal||0)+(a.pipeline||0)+(a.timeline||0)+(a.properties||0);
    a.catErr_email     = (a.emailSent||0)+(a.emailHub||0)+(a.profTone||0)+(a.signature||0);
    a.catErr_whatsapp  = (a.waUsed||0)+(a.waLogged||0)+(a.waNote||0);
    a.catErr_task      = (a.taskCreated||0)+(a.taskDone||0)+(a.taskType||0);
    a.errorLinks       = errorLinks[a.name] || {};
    a.trend            = [];
    a.mismatchQualNoDeal = 0; a.mismatchDealWrongStage = 0; a.mismatchQualWrongStage = 0;
    a.caseQualified = 0; a.caseNoAnswer = 0; a.caseCannotDial = 0;
    a.qualPct = null; a.naPct = null;
    return a;
  });
}


    

      ✅ What this does: The buildErrorLinks() function reads your RAW AUDIT sheet, auto-detects every column from its header name, and for each audit row that has an error, saves the case link + contact name + date. These are bundled into the JSON response as errorLinks per agent — the dashboard reads them instantly, no sheet-sharing required.
